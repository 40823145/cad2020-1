有些情况我们需要通过系统的质量属性获取一些信息.

这个api帮助里自带例子.

  1.         private void btn_GetMass_Click(object sender, EventArgs e)
  2.         {
  3.             // 获取质量属性可参考 Get Mass Properties of Visible and Hidden Components Example (C#)
  4.  
  5.             ISldWorks swApp = Utility.ConnectToSolidWorks();
  6.  
  7.             ModelDoc2 swModel = swApp.ActiveDoc;
  8.  
  9.             ModelDocExtension swModelDocExt = (ModelDocExtension)swModel.Extension;
  10.  
  11.             swModelDocExt.IncludeMassPropertiesOfHiddenBodies = false;
  12.             int massStatus = 0;
  13.  
  14.             double[] massProperties = (double[])swModelDocExt.GetMassProperties(1, ref massStatus);
  15.             if ((massProperties != null))
  16.             {
  17.                 Debug.Print(" CenterOfMassX = " + massProperties[0]);
  18.                 Debug.Print(" CenterOfMassY = " + massProperties[1]);
  19.                 Debug.Print(" CenterOfMassZ = " + massProperties[2]);
  20.                 Debug.Print(" Volume = " + massProperties[3]);
  21.                 Debug.Print(" Area = " + massProperties[4]);
  22.                 Debug.Print(" Mass = " + massProperties[5]);
  23.                 Debug.Print(" MomXX = " + massProperties[6]);
  24.                 Debug.Print(" MomYY = " + massProperties[7]);
  25.                 Debug.Print(" MomZZ = " + massProperties[8]);
  26.                 Debug.Print(" MomXY = " + massProperties[9]);
  27.                 Debug.Print(" MomZX = " + massProperties[10]);
  28.                 Debug.Print(" MomYZ = " + massProperties[11]);
  29.             }
  30.             Debug.Print("-------------------------------");
  31.         }
